1. Write a JavaScript program to display the current day and time in the following format. Sample Output : Today is : Tuesday. Current time is : 10 PM : 30 : 38
Problem Solution Code:
function GetDayandTime() { var today= new Date(); var day = today.getDay(); var dayList = ["Sunday","Monday","Tuesday","Wednesday","Thursday","Friday","Saturday "]; // Get the current time let hours = today.getHours(); const ampm = hours >= 12 ? "PM" : "AM"; hours = hours % 12 || 12; const minutes = today.getMinutes(); const seconds = today.getSeconds(); console.log("Today Is:" + dayList[day] ); console.log("Current time is: ", hours, ampm + " : ", minutes, " : ", seconds); } GetDayandTime();

Today Is:Thursday Current time is: 6 PM : 33 : 29

4. Write a JavaScript program to compute the sum of the two given integers. If the two values are the same, then return triple their sum.
Problem Solution Code:
function sumNumber (x, y) { if (x == y) { return 3 * (x + y); } else { return (x + y); } } console.log(sumNumber(30, 20)); console.log(sumNumber(20, 20));
Output:
50 120

6. Write a JavaScript program to create another string by adding "Py" in front of a given string. If the given string begins with "Py" return the original string.
Problem Solution Code:
function addPy(str) { if (str.startsWith("Py")) { return str; } else { return "Py" + str; } } console.log(addPy("thon")); console.log(addPy("Python"));
Output:
Python Python

8.Write a JavaScript program to compute the sum of all the digits that occur in a given string.
Problem Solution Code:
function sumDigits(str) { var sum = 0; for (var i = 0; i < str.length; i++) { var char = str.charAt(i); if (!isNaN(char)) { sum += parseInt(char); } } return sum; } var inputString = "Hello9733World9749"; var digitSum = sumDigits(inputString); console.log(digitSum);

51
9. Write a JavaScript program to break an URL address and put its parts into an array. Note: url structure : ://.org[/] and there may be no part in the address.
Problem Solution Code:
function breakURL(url) { var urlObject = new URL(url); var parts = [urlObject.protocol, urlObject.host, urlObject.pathname, urlObject.search, urlObject.hash]; return parts; } var url = "https://www.w3schools.com/jsref/jsref_isnan.asp"; var urlParts = breakURL(url); console.log(urlParts);
Output:
[ 'https:', 'www.w3schools.com', '/jsref/jsref_isnan.asp', '', '' ]
10. Write a JavaScript program to sort the strings of a given array of strings in order of increasing length. Note: Do not change the order if the lengths of two string are same.
Problem Solution Code:
function sortStrings(strings) { strings.sort(function(a, b) { return a.length - b.length; }); return strings; } var arr = ["apple", "banana", "cherry", "date", "elderberry"]; var sortedArr = sortStrings(arr); console.log(sortedArr);
Output:
[ 'date', 'apple', 'banana', 'cherry', 'elderberry' ]
